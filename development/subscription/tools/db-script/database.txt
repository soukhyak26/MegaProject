var now=new Date();
for (var i = 0; i < 6; i++) {
  var yesterdayMs = new Date (now.getTime() + 1000*60*60*24*i);
  var month = (yesterdayMs.getMonth() + 1);
  var day = (yesterdayMs.getDate());
  var year = (yesterdayMs.getFullYear());
  var date = year + "-" + month + "-" + day;
  var dateTime = date + "-03:42:55";
  db.sysDateView.save({ "_id" : 1, "currentDate" : date });
  db.sysDateTimeView.save({ "_id" : 1, "currentDate" : dateTime });
  sleep(200);
}
